Service Implementation 
@Service
public class SmsService {
    
    private final TwilioConfig twilioConfig;
    private final RateLimiter rateLimiter;
    
    public SmsService(TwilioConfig twilioConfig, 
                      @Value("${sms.rate.limit}") int rateLimit) {
        this.twilioConfig = twilioConfig;
        this.rateLimiter = RateLimiter.create(rateLimit);
        Twilio.init(twilioConfig.getAccountSid(), 
                    twilioConfig.getAuthToken());
    }
    
    @PreAuthorize("hasPermission(#request, 'SEND_SMS')")
    public String sendSms(SmsRequest request) {
        // Check rate limit
        if (!rateLimiter.tryAcquire()) {
            throw new RateLimitExceededException(
                "SMS rate limit exceeded");
        }
        
        // Validate recipient consent
        if (!consentService.hasConsent(request.getPhone())) {
            throw new ConsentException(
                "Recipient has not consented to receive SMS");
        }
        
        // Send message
        Message message = Message.creator(
            new PhoneNumber(request.getPhone()),
            new PhoneNumber(twilioConfig.getPhoneNumber()),
            request.getMessage()
        ).create();
        
        return message.getSid();
    }
}
Security Endpoint 
@RestController
@RequestMapping("/api/sms")
public class SmsController {
    
    private final SmsService smsService;
    
    public SmsController(SmsService smsService) {
        this.smsService = smsService;
    }
    
    @PostMapping("/send")
    public ResponseEntity<SmsResponse> sendSms(
            @RequestBody @Valid SmsRequest request) {
        String messageId = smsService.sendSms(request);
        return ResponseEntity.ok(new SmsResponse(
            "SMS sent successfully", messageId));
    }
    
    @PostMapping("/opt-out")
    public ResponseEntity<String> handleOptOut(
            @RequestParam String phone) {
        consentService.revokeConsent(phone);
        return ResponseEntity.ok(
            "Opt-out request processed. No further messages will be sent.");
    }
}

Best Practices

  • Always obtain explicit consent
  • Provide clear opt-out instructions
  • Respect all opt-out requests immediately
  • Limit message frequency
  • Identify your business clearly
